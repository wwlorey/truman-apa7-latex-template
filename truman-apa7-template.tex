% CONFIGURATION

\documentclass[12pt]{article}

\usepackage[left=1in,right=1in,top=1in,bottom=1in]{geometry}
\usepackage{indentfirst} % Indent the first sentence in each paragraph.
\usepackage{lipsum}      % Lorem Ipsum generation
\usepackage{setspace}    % Allow the use of custom spacing blocks.
\usepackage{titling}     % Allow formatting of the title page.

% TODO: Define paper metadata.
\newcommand{\papertitle}{Paper Title}
\author{Author}
\newcommand{\coursetitle}{Course Title}
\newcommand{\instructor}{Instructor}
\newcommand{\duedate}{Due date}
% The `\thetitle` command doesn't consistently return the defined title so the title has also been saved in `\papertitle`.
\title{\papertitle}

% Render the page number in upper right hand side of the header.
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}                         % Clear existing header and footer.
\setlength{\headheight}{15pt}      % With a font size of 12pt, the compiler complains about header heights less than this.
\renewcommand{\headrulewidth}{0pt} % Hide the header's horizontal rule.
\fancyhead[HR]{\thepage}           % HR = header right

\usepackage{fontspec}
\setmainfont{Times New Roman}

\linespread{2}

% Format the section titles.
\usepackage{titlesec}
\titleformat{\section}{\normalfont\normalsize\bfseries\center}{}{0pt}{}
\titlespacing*{\section}{0pt}{0pt}{0pt}

\titleformat{\subsection}{\normalfont\normalsize\bfseries}{}{0pt}{}
\titlespacing*{\subsection}{0pt}{0pt}{0pt}

\titleformat{\subsubsection}[runin]{\normalfont\normalsize\bfseries}{}{.5in}{}
\titlespacing*{\subsubsection}{0pt}{0pt}{5pt}

\renewcommand{\maketitle}{
  \begin{center}
    \vspace*{150pt}

    \begin{singlespace}
      \papertitle

      \theauthor
    \end{singlespace}

    \vspace{50pt}

    \textit{A paper submitted in partial}

    \textit{fulfillment of the requirements of COUN XXX}

    \textit{\coursetitle}

    \textit{\instructor}

    \textit{\duedate}
  \end{center}
}

\begin{document}


% TITLE PAGE

\maketitle

\newpage


% ABSTRACT

\setlength{\parindent}{0in}

\newpage


% BODY OF PAPER

\section{\papertitle}

\lipsum[2]

\subsection{Subsection}

\lipsum[5]

\lipsum[3]

\subsubsection{Subsubsection}

\lipsum[5]

\lipsum[3]

\newpage
\section{References}
